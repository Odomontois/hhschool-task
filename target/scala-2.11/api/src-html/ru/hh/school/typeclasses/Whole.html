006package ru.hh.school.typeclasses;
007
008import ru.hh.school.utils.Pair;
009import ru.hh.school.utils.Unfold;
010
011import java.util.LinkedList;
012import java.util.List;
013
014/**
015 * Интерфейс, отвечающий за операции над целыми числами (Int, Long, BigInteger...)
016 * Эквивалент Integral в scala\haskell
017 *
018 * @param <N>
019 */
020public interface Whole<N> extends Numeric<N>, Ordering<N> {
021    N quot(N x, N y);
022
023    N mod(N x, N y);
024
025
026    default int sign(N x) {
027        final int cmp = compare(x, zero());
028        if (cmp > 0) return 1;
029        else if (cmp == 0) return 0;
030        else return -1;
031    }
032
033    /**
034     * Имплементация по умолчанию методом Эйлера
035     * @return Наибольший общий делитель x и y
036     */
037    default N gcd(N x, N y) {
038        if (compare(x, y) > 0) return gcd(y, x);
039        if (compare(x, zero()) == 0) return y;
040        return gcd(mod(y, x), x);
041    }
042
043    default Pair<N, N> divMod(N x, N y) {
044        final N div = quot(x, y);
045        final N rem = mod(x, y);
046        return Pair.of(div, rem);
047    }
048
049    default List<N> digits(N num, N base) {
050        final LinkedList<N> result = new LinkedList<>();
051
052        Unfold.of(n -> !n.equals(zero()), n -> divMod(n, base).swap(), num).forEach(result::addFirst);
053
054        return result;
055    }
056
057    default String toString(Iterable<N> digits, N base) {
058        final boolean brackets = compare(base, fromInt(36)) > 0;
059
060        final StringBuilder sb = new StringBuilder();
061
062        for (N dig : digits) {
063            if (brackets) {
064                sb.append("[");
065                sb.append(dig.toString());
066                sb.append("]");
067            } else {
068                sb.append(DIGITS.charAt(toInt(dig)));
069            }
070        }
071
072        return sb.toString();
073    }
074
075    String DIGITS = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
076}
