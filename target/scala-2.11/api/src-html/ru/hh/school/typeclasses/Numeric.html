006package ru.hh.school.typeclasses;
007
008/**
009 * Операции сложения, умножения вычитания, задающие кольцо
010 * Аналог Num в haskell, Numeric в scala
011 * @param <N>
012 */
013public interface Numeric<N> {
014    N add(N x, N y);
015
016    N subtract(N x, N y);
017
018    N multiply(N x, N y);
019
020    N fromInt(int x);
021
022    int toInt(N x);
023
024    N parse(String s);
025
026    int sign(N x);
027
028    default N zero() {
029        return fromInt(0);
030    }
031
032    default N one() {
033        return fromInt(1);
034    }
035
036    default N abs(N x) {
037        return multiply(x, fromInt(sign(x)));
038    }
039
040    default N negate(N x) {
041        return multiply(x, fromInt(-1));
042    }
043
044    default N pow(N x, int p) {
045        if (p == 0) return one();
046        final N higher = pow(multiply(x, x), p / 2);
047        return p % 2 == 0 ? higher : multiply(x, higher);
048    }
049}
