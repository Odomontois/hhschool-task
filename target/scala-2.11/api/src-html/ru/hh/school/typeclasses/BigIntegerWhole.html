006package ru.hh.school.typeclasses;
007
008import ru.hh.school.utils.Pair;
009
010import java.math.BigInteger;
011
012public class BigIntegerWhole implements Whole<BigInteger> {
013    private BigIntegerWhole() {
014    }
015
016    public static Whole<BigInteger> INSTANCE = new BigIntegerWhole();
017
018    @Override
019    public BigInteger add(BigInteger x, BigInteger y) {
020        return x.add(y);
021    }
022
023    @Override
024    public BigInteger subtract(BigInteger x, BigInteger y) {
025        return x.subtract(y);
026    }
027
028    @Override
029    public BigInteger multiply(BigInteger x, BigInteger y) {
030        return x.multiply(y);
031    }
032
033    @Override
034    public BigInteger quot(BigInteger x, BigInteger y) {
035        return x.divide(y);
036    }
037
038    @Override
039    public BigInteger mod(BigInteger x, BigInteger y) {
040        return x.mod(y);
041    }
042
043    @Override
044    public BigInteger fromInt(int x) {
045        return BigInteger.valueOf(x);
046    }
047
048    @Override
049    public BigInteger zero() {
050        return BigInteger.ZERO;
051    }
052
053    @Override
054    public BigInteger one() {
055        return BigInteger.ONE;
056    }
057
058    @Override
059    public Pair<BigInteger, BigInteger> divMod(BigInteger x, BigInteger y) {
060        final BigInteger[] divmod = x.divideAndRemainder(y);
061        return Pair.of(divmod[0], divmod[1]);
062    }
063
064    @Override
065    public BigInteger abs(BigInteger x) {
066        return x.abs();
067    }
068
069    @Override
070    public int sign(BigInteger x) {
071        return x.signum();
072    }
073
074    @Override
075    public int toInt(BigInteger x) {
076        return x.intValue();
077    }
078
079    @Override
080    public int compare(BigInteger x, BigInteger y) {
081        return x.compareTo(y);
082    }
083
084    @Override
085    public BigInteger gcd(BigInteger x, BigInteger y) {
086        return x.gcd(y);
087    }
088
089    @Override
090    public BigInteger negate(BigInteger x) {
091        return x.negate();
092    }
093
094    @Override
095    public BigInteger pow(BigInteger x, int p) {
096        return x.pow(p);
097    }
098
099    @Override
100    public BigInteger parse(String s) {
101        return new BigInteger(s);
102    }
103}
