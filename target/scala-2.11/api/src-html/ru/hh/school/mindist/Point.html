006package ru.hh.school.mindist;
007
008import java.util.Comparator;
009
010/**
011 * Точка на двумерной плоскости
012 */
013public class Point {
014    protected double x, y;
015
016    public Point(double x, double y) {
017        this.x = x;
018        this.y = y;
019    }
020
021    public double getX() {
022        return x;
023    }
024
025    public double getY() {
026        return y;
027    }
028
029    @Override
030    public String toString() {
031        return '(' + x + " , " + y + ')';
032    }
033
034    public static int compareXthenY(Point p1, Point p2) {
035        return p1.x == p2.x ? Double.compare(p1.y, p2.y) : Double.compare(p1.x, p2.x);
036    }
037
038    public static int compareYthenX(Point p1, Point p2) {
039        return p1.y == p2.y ? Double.compare(p1.x, p2.x) : Double.compare(p1.y, p2.y);
040    }
041
042    public double dist(Point point) {
043        final double dx = point.x - this.x;
044        final double dy = point.y - this.y;
045        return Math.sqrt(dx * dx + dy * dy);
046    }
047
048    public int compareDistance(Point p1, Point p2) {
049        return Double.compare(this.dist(p1), this.dist(p2));
050    }
051
052    @Override
053    public boolean equals(Object o) {
054        if (this == o) return true;
055        if (!(o instanceof Point)) return false;
056
057        Point point = (Point) o;
058
059        return Double.compare(point.x, x) == 0 && Double.compare(point.y, y) == 0;
060    }
061
062    @Override
063    public int hashCode() {
064        int result;
065        long temp;
066        temp = Double.doubleToLongBits(x);
067        result = (int) (temp ^ (temp >>> 32));
068        temp = Double.doubleToLongBits(y);
069        result = 31 * result + (int) (temp ^ (temp >>> 32));
070        return result;
071    }
072}
