006package ru.hh.school.mindist;
007
008public class LabelledPoint<L> extends Point {
009    final L label;
010
011    public LabelledPoint(double x, double y, L label) {
012        super(x, y);
013        this.label = label;
014    }
015
016    @Override
017    public String toString() {
018        return String.format("[%s](%s,%s)", label, x, y);
019    }
020
021    @Override
022    public boolean equals(Object o) {
023        if (this == o) return true;
024        if (!(o instanceof LabelledPoint)) return false;
025        if (!super.equals(o)) return false;
026
027        LabelledPoint<?> that = (LabelledPoint<?>) o;
028
029        return !(label != null ? !label.equals(that.label) : that.label != null);
030
031    }
032
033    @Override
034    public int hashCode() {
035        int result = super.hashCode();
036        result = 31 * result + (label != null ? label.hashCode() : 0);
037        return result;
038    }
039}
