006package ru.hh.school.runnable;
007
008import ru.hh.school.typeclasses.BigIntegerWhole;
009import ru.hh.school.fraction.Fraction;
010import ru.hh.school.typeclasses.LongWhole;
011import ru.hh.school.typeclasses.Whole;
012import ru.hh.school.utils.CyclicSequence;
013import ru.hh.school.utils.Pair;
014
015import java.io.IOException;
016import java.nio.file.Files;
017import java.nio.file.Paths;
018import java.util.*;
019
020public class Fractions {
021    enum Flag implements RunFlag {
022        BIG_INTEGER("-b", "--bigint"),
023        LONG("-l", "--long"),
024        CONSTANT_MEMORY("-cm", "--O1memory"),
025        MEMOIZE("-s", "--memoize");
026
027        private String[] params;
028
029        Flag(String... params) {
030            this.params = params;
031        }
032
033        @Override
034        public String[] getParams() {
035            return params;
036        }
037    }
038
039    static <N> Pair<Fraction<N>, N> parseFraction(String line, Whole<N> whole) {
040        final StringTokenizer tok = new StringTokenizer(line);
041        final N numerator = whole.parse(tok.nextToken());
042        final N denominator = whole.parse(tok.nextToken());
043        final N base = whole.parse(tok.nextToken());
044        final Fraction<N> fraction = Fraction.create(numerator, denominator, whole);
045        return Pair.of(fraction, base);
046    }
047
048
049    static <N> List<Pair<Fraction<N>, N>> parseFractions(String filename, Whole<N> whole) throws IOException {
050        final List<String> lines = Files.readAllLines(Paths.get(filename));
051        final List<Pair<Fraction<N>, N>> fractions = new ArrayList<>(lines.size());
052
053        lines.stream().map(line -> parseFraction(line, whole)).forEach(fractions::add);
054
055        return fractions;
056    }
057
058    static <N> List<String> createReprs(String filename, Whole<N> whole, CyclicSequence.Method method) throws IOException {
059        final List<Pair<Fraction<N>, N>> fractions = parseFractions(filename, whole);
060        final List<String> reprs = new ArrayList<>(fractions.size());
061
062        fractions.stream()
063                .map(fraction -> fraction.getFirst().toDigitalString(fraction.getSecond(), method))
064                .forEach(reprs::add);
065
066        return reprs;
067    }
068
069    public static void main(String[] args) {
070        try {
071            final Set<Flag> flags = RunFlag.parse(Flag.values(), Arrays.asList(args).subList(1, args.length));
072            final Whole<?> whole;
073            final CyclicSequence.Method method;
074
075            //BigInteger приоритетней и по умолчанию
076            if (flags.contains(Flag.BIG_INTEGER))
077                whole = BigIntegerWhole.INSTANCE;
078            else if (flags.contains(Flag.LONG))
079                whole = LongWhole.INSTANCE;
080            else whole = BigIntegerWhole.INSTANCE;
081
082            //Constant-memory приоритетней и по умолчанию
083            if (flags.contains(Flag.CONSTANT_MEMORY))
084                method = CyclicSequence.Method.CONSTANT_MEMORY;
085            else if (flags.contains(Flag.MEMOIZE))
086                method = CyclicSequence.Method.MEMOIZE;
087            else method = CyclicSequence.Method.MEMOIZE;
088
089            final List<String> reprs = createReprs(args[0], whole, method);
090
091            reprs.forEach(System.out::println);
092        } catch (IOException ex) {
093            System.err.println(ex.getMessage());
094        } catch (Throwable ex) {
095            ex.printStackTrace(System.err);
096        }
097    }
098}
