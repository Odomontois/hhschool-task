006package ru.hh.school.utils;
007
008import java.util.*;
009
010public class CyclicSequence<A> {
011    final Iterable<A> elements;
012    final List<A> prefix;
013    final List<A> cycle;
014
015    public enum Method {
016        /**
017         * Метод, находящий цикл за O(префикс + длина цикла) памяти,
018         * с использованием минимального числа итераций
019         * в случае, если для элемента определены адекватные equals\hashCode
020         */
021        MEMOIZE {
022            @Override
023            <A> Pair<Integer, Integer> findCycle(Iterable<A> elements) {
024                final LinkedHashSet<A> seen = new LinkedHashSet<>();
025                final Iterator<A> it = elements.iterator();
026                A current = it.next();
027
028                while (!seen.contains(current)) {
029                    seen.add(current);
030                    current = it.next();
031                }
032
033                int prefix = 0;
034                for (A elem : seen) {
035                    if (elem.equals(current)) break;
036                    prefix += 1;
037                }
038                return Pair.of(prefix, seen.size() - prefix);
039            }
040        },
041
042        /**
043         * Метод, находящий цикл за O(1) памяти
044         * в случае, если итератор использует O(1) памяти
045         */
046        CONSTANT_MEMORY {
047            @Override
048            <A> Pair<Integer, Integer> findCycle(Iterable<A> elements) {
049                final Iterator<A>
050                        slow = elements.iterator(),
051                        fast = elements.iterator();
052                int meet = 1;
053                fast.next();
054                while (!(slow.next().equals(fast.next()))) {
055                    meet++;
056                    fast.next();
057                }
058
059                int prefix = 0;
060                final Iterator<A>
061                        fromStart = elements.iterator(),
062                        skipped = elements.iterator();
063                for (int i = 0; i < meet; i++)
064                    skipped.next();
065                while (!fromStart.next().equals(skipped.next()))
066                    prefix++;
067
068                int cycle = 1;
069                final A current = fromStart.next();
070                while (!fromStart.next().equals(current)) {
071                    cycle++;
072                }
073
074                return Pair.of(prefix, cycle);
075            }
076        };
077
078        abstract <A> Pair<Integer, Integer> findCycle(Iterable<A> elements);
079    }
080
081    protected CyclicSequence(Iterable<A> elements, List<A> prefix, List<A> cycle) {
082        this.elements = elements;
083        this.prefix = prefix;
084        this.cycle = cycle;
085    }
086
087    public static <A> CyclicSequence<A> create(Iterable<A> elements, Method method) {
088        final Pair<Integer, Integer> cycle = method.findCycle(elements);
089
090        return create(elements, cycle.getFirst(), cycle.getSecond());
091    }
092
093    public static <A> CyclicSequence<A> create(Iterable<A> elements, int prefixLength, int cycleLength) {
094        final int size = prefixLength + cycleLength;
095        final List<A> full = new ArrayList<>(size);
096        final Iterator<A> it = elements.iterator();
097        for (int i = 0; i < size; i++)
098            full.add(it.next());
099
100        return new CyclicSequence<>(elements,
101                full.subList(0, prefixLength),
102                full.subList(prefixLength, size));
103    }
104
105    public Iterable<A> getElements() {
106        return elements;
107    }
108
109    public List<A> getPrefix() {
110        return prefix;
111    }
112
113    public List<A> getCycle() {
114        return cycle;
115    }
116}
