006package ru.hh.school.utils;
007
008import java.util.*;
009
010/**
011 * Двусвязный список с поддержикой операции слияния
012 * производящую из двух сортированых списков новый
013 */
014public class SortedList<E> extends LinkedList<E> {
015    /**
016     * Слияние заранее сортированных коллекций в новую сортированную коллекцию
017     * за O(M+N)
018     */
019    public SortedList<E> merge(SortedList<E> list2, Comparator<E> comp) {
020        SortedList<E> result = new SortedList<>();
021        PeekableIterator<E> it1 = new PeekableIterator<>(this.iterator());
022        PeekableIterator<E> it2 = new PeekableIterator<>(list2.iterator());
023
024        while (it1.hasNext() && it2.hasNext()) {
025            if (comp.compare(it1.peekNext(), it2.peekNext()) <= 0) {
026                result.addLast(it1.next());
027            } else result.addLast(it2.next());
028        }
029
030        it1.forEachRemaining(result::addLast);
031        it2.forEachRemaining(result::addLast);
032        return result;
033    }
034
035    public static <E> SortedList<E> singleton(E e) {
036        final SortedList<E> result = new SortedList<>();
037        result.add(e);
038        return result;
039    }
040
041    public static <E extends Comparable<E>> SortedList<E> from(Collection<E> es) {
042        return new SortedList<>(es, Comparator.<E>naturalOrder());
043    }
044
045    public SortedList(Collection<? extends E> collection, Comparator<E> comparator) {
046        super(collection);
047        this.sort(comparator);
048    }
049
050    public SortedList() {
051        super();
052    }
053
054
055}
