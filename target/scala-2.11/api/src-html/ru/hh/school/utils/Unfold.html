006package ru.hh.school.utils;
007
008import java.util.Iterator;
009import java.util.NoSuchElementException;
010import java.util.function.Function;
011
012/**
013 * Аналог Data.List.unfoldr в Haskell
014 * Представляет из себя последовательность элементов, генерируемую вызовом бинарной функции
015 */
016public class Unfold<A, B> implements Iterable<A> {
017    final private Function<B, Pair<A, B>> unfold;
018    final private B start;
019    final private Function<B, Boolean> doWhile;
020
021    protected Unfold(Function<B, Pair<A, B>> unfold, B start, Function<B, Boolean> doWhile) {
022        this.unfold = unfold;
023        this.start = start;
024        this.doWhile = doWhile;
025    }
026
027    /**
028     * Генерация последовательности на основе промежуточного состояния
029     */
030    public static <A, B> Unfold<A, B> of(Function<B, Boolean> hasNext, Function<B, Pair<A, B>> unfold, B start) {
031        return new Unfold<>(unfold, start, hasNext);
032    }
033
034    /**
035     * Упрощённый вариант - для функций, генерирующих последовательность элеметов на основе эндоморфизмов
036     */
037    public static <A> Unfold<A, A> endo(Function<A, Boolean> hasNext, Function<A, A> unfold, A start) {
038        return new Unfold<>(unfold.andThen(x -> Pair.of(x, x)), start, hasNext);
039    }
040
041
042    @Override
043    public Iterator<A> iterator() {
044        return new UnfoldIterator(start);
045    }
046
047
048    private class UnfoldIterator implements Iterator<A> {
049        private B state;
050
051        public UnfoldIterator(B state) {
052            this.state = state;
053        }
054
055        @Override
056        public boolean hasNext() {
057            return doWhile.apply(state);
058        }
059
060        @Override
061        public A next() {
062            if (!hasNext()) throw new NoSuchElementException();
063            final Pair<A, B> result = unfold.apply(state);
064            this.state = result.getSecond();
065            return result.getFirst();
066        }
067    }
068}
